### 1. Database Constraints (SQL)
We need a way to track how many active ads a user currently has and how many images are attached to each.

```sql
-- Security Check: Prevent more than 200 ads per user
CREATE OR REPLACE FUNCTION check_user_listing_limit()
RETURNS TRIGGER AS $$
BEGIN
  IF (SELECT COUNT(*) FROM listings WHERE seller_id = NEW.seller_id AND status = 'active') >= 200 THEN
    RAISE EXCEPTION 'Maximum limit of 200 active listings reached.';
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_limit_listings
BEFORE INSERT ON listings
FOR EACH ROW EXECUTE FUNCTION check_user_listing_limit();

-- Image Constraint: Ensure the array does not exceed 4 items
ALTER TABLE listings 
ADD CONSTRAINT check_max_images 
CHECK (cardinality(images) <= 4);
```
