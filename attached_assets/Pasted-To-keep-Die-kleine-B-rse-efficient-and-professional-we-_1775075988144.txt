### **2. Admin Approval View (The "Welik" Dashboard)**
This view is strictly protected. Only you (`welik.jakob@gmail.com`) can see it. It fetches all ads where `payment_status` is `'pending'` and allows a one-click activation.

```tsx
import { useState, useEffect } from 'react';
import { supabase } from '../lib/supabaseClient';

export default function AdminApprovalView() {
  const [pendingListings, setPendingListings] = useState([]);
  const [loading, setLoading] = useState(true);

  // 1. Fetch only listings waiting for payment confirmation
  useEffect(() => {
    fetchPending();
  }, []);

  async function fetchPending() {
    const { data } = await supabase
      .from('listings')
      .select('*, profiles(username)')
      .eq('payment_status', 'pending')
      .order('created_at', { ascending: false });
    setPendingListings(data || []);
    setLoading(false);
  }

  // 2. The "One-Click" Activation Logic
  async function approveListing(id) {
    const { error } = await supabase
      .from('listings')
      .update({ payment_status: 'completed', status: 'active' })
      .eq('id', id);

    if (!error) {
      setPendingListings(prev => prev.filter(item => item.id !== id));
      alert('Listing activated and live on "Die kleine Börse"!');
    }
  }

  return (
    <div className="p-10 bg-slate-50 min-h-screen font-sans">
      <div className="max-w-5xl mx-auto">
        <header className="flex justify-between items-end mb-10">
          <div>
            <h1 className="text-4xl font-light text-slate-900">Payment Approvals</h1>
            <p className="text-slate-500 mt-2">Logged in as: <span className="font-semibold text-slate-900">welik.jakob@gmail.com</span></p>
          </div>
          <div className="bg-white px-6 py-3 rounded-2xl shadow-sm border border-slate-100">
            <span className="text-xs uppercase text-slate-400 font-bold tracking-widest">Queue</span>
            <p className="text-xl font-medium">{pendingListings.length} Ads</p>
          </div>
        </header>

        {loading ? <p>Loading vault...</p> : (
          <div className="grid gap-4">
            {pendingListings.map(item => (
              <div key={item.id} className="bg-white p-6 rounded-[2rem] border border-slate-100 flex justify-between items-center hover:shadow-md transition-shadow">
                <div className="flex gap-6 items-center">
                  <div className="h-16 w-16 bg-slate-100 rounded-2xl flex items-center justify-center text-2xl">
                    {item.category === 'vehicles-mobility' ? '🏎️' : '🏢'}
                  </div>
                  <div>
                    <h3 className="font-semibold text-lg text-slate-900">{item.title}</h3>
                    <p className="text-sm text-slate-500">
                      User: <span className="text-slate-700">{item.profiles?.username}</span> • 
                      Fee Due: <span className="font-bold text-slate-900">€{item.listing_fee}</span>
                    </p>
                  </div>
                </div>

                <div className="flex gap-3">
                  <button className="px-6 py-3 text-sm font-medium text-slate-400 hover:text-red-500">Reject</button>
                  <button 
                    onClick={() => approveListing(item.id)}
                    className="bg-slate-900 text-white px-8 py-3 rounded-xl text-sm font-semibold hover:bg-blue-600 transition-colors"
                  >
                    Confirm Payment & Go Live
                  </button>
                </div>
              </div>
            ))}
            {pendingListings.length === 0 && (
              <div className="text-center py-20 bg-white rounded-[3rem] border border-dashed border-slate-200">
                <p className="text-slate-400 uppercase tracking-widest text-xs">All clear. No pending payments.</p>
              </div>
            )}
          </div>
        )}
      </div>
    </div>
  );
}
```
